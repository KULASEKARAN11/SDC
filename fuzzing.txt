#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

void vulnerable_function(char *input) {
    char buffer[10];  // small buffer

    // Unsafe copy (vulnerability)
    strcpy(buffer, input);

    printf("Processed: %s\n", buffer);
}

void generate_random_input(char *input, int length) {
    for (int i = 0; i < length; i++) {
        input[i] = (rand() % 94) + 33; // printable ASCII
    }
    input[length] = '\0';
}

int main() {
    srand(time(NULL));

    char input[100];

    for (int i = 0; i < 100; i++) {
        int len = rand() % 50;  // random length

        generate_random_input(input, len);

        printf("Test %d: %s\n", i + 1, input);

        vulnerable_function(input);
    }

    return 0;
}
